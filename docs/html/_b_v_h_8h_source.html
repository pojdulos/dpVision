54 bool rayIntersectsTriangle(const Ray& ray,

57 const float _EPSILON = 1e-6f; // Tolerancja numeryczna
58
59 // Krawêdzie trójk¹ta
60 CVector3f E1 = V1 - V0;
61 CVector3f E2 = V2 - V0;

63 // Wektor normalny do p³aszczyzny trójk¹ta
64 CVector3f P = ray.direction.crossProduct(E2);

66 // Wyznacznik macierzy (determinant)
67 float det = E1.dotProduct(P);
68
69 // Jeœli det jest bliski zeru, promieñ jest równoleg³y do p³aszczyzny trójk¹ta
70 if (fabs(det) < _EPSILON) return false;
71

74 // Wektor od V0 do pocz¹tku promienia
75 CVector3f T = ray.origin - V0;
76
77 // Oblicz wspó³czynnik u (po³o¿enie barycentryczne)
78 u = T.dotProduct(P) * invDet;
79 if (u < 0.0f || u > 1.0f) return false;
80
81 // Oblicz wspó³czynnik v
82 CVector3f Q = T.crossProduct(E1);
83 v = ray.direction.dotProduct(Q) * invDet;
84 if (v < 0.0f || u + v > 1.0f) return false;
